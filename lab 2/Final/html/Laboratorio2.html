vbeta = [0:0.25:1];

for beta = vbeta

    T1 = 1+beta;
    T2= alpha * T1;
    U1 = -2/(T1 * (1 + alpha)); % Relação descoberta analiticamente entre U1 e T1
    U2 = -U1;

    [u1,t1] = rcos(T1,beta);
